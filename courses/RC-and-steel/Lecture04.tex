			\item[하우 트러스(howe truss)] 상현재는 압축, 하현재는 인장에 저항하며, 사재는 주로 압축, 수직재는 인장에 저항하는 트러스
			\item[와렌 트러스(warren truss)] 수직재가 없는 경우, 타 트러스에 비하여 부재수가 적고 구조가 간단하며 연속교량 트러스에 많이 사용되나, 현재의 길이가 과대하여 강성이 낮음. 이를 보완하기 위하여 수직재를 사용하는 경우도 있음
		\end{description}}

	\begin{frame}{트러스의 판별}
		1) 안정과 불안정: 외력이 작용하는 경우 구조물의 평형상태로 판별
		\begin{description}
			\item[내적 안정] 외력에 의해 구조물이 \alert{변형}되지 않는 경우
			\item[내적 불안정] 외력에 의해 구조물이 변형되는 경우
			\item[외적 안정] 외력에 의해 구조물이 \alert{이동}하지 않는 경우
			\item[외적 불안정] 외력에 의해 구조물이 이동되는 경
		\end{description}
		\begin{figure}
			\includegraphics[height=0.30\textheight]{03STABILITY}
			\caption{내적안정/외적안정(좌); 내적불안정/외적안정(중); 내적안정/외적불안정(우)}
		\end{figure}		
	\end{frame}
	\begin{frame}{트러스의 판별}
		2) 정정과 부정정: 힘의 평형방정식으로 반력과 부재를 구할 수 있는가로 판별
		\begin{description}
			\item[정정] 힘의 평형방정식만으로 반력을 구할 수 있는 구조물
			\item[부정정] 힘의 평형방정식만으로는 반력을 구할 수 없는 구조물
		\end{description}
		\begin{figure}
			\includegraphics[height=0.30\textheight]{03STABILITY02}
			\caption{정정구조물(좌)과 부정정구조물(우)}
		\end{figure}		
	\end{frame}
	\begin{frame}{트러스의 판별}
		\begin{description}
			\item[$b+r<2j$] 불안정
			\item[$b+r\geq 2j$] 반력이 불충분하거나 트러스 요소 중 하나라도 내적불안정할 경우 불안정. 안정한 트러스에 대해 $b+r=2j$인 경우 정정, $b+r>2j$인 경우 부정정.
		\end{description}	
	\end{frame}

	\begin{frame}{트러스 해법 상의 가정}
		\begin{itemize}
			\item 트러스 부재와 작용 외력은 동일평면에 있다. 
			\item 절점은 전혀 마찰이 없는 힌지로 생각한다. 
			\item 외력은 모두 절점에만 집중하중으로 작용한다. 
			\item 절점과 절점을 연결한 직선은 부재축과 일치한다. 
			\item 하중이 작용한 경우에도 절점의 위치는 변하지 않는다고 가정한다. 
		\end{itemize}
	\end{frame}
