			\alert{장점}
			\begin{enumerate}
				\item 단위면적 당 강도가 크다.
				\item 인성이 커서 변형에 유리하고 소성변형능력이 우수하다. 
				\item 재료가 균질하다. 
				\item 세장한 부재가 가능하다.
				\item 공사기간이 빠르다. 
				\item 기존 건축물의 증축, 보수가 용이하다.
				\item 환경친화적인 재료이다. 
				\item 하이테크 건축재료이다. 
			\end{enumerate}
			\column{.49\linewidth}
			\alert{단점}
			\begin{enumerate}
				\item 내화성을 고려해야 한다.
				\item 부식에 따른 유지관리가 필요하다. 
				\item 접합부의 신중한 설계와 용접부의 검사가 필요하다. 
				\item 좌굴의 영향이 크다. 
				\item 처짐 및 진동을 고려해야 한다. 
				\item 응력반복에 따른 피로에 의해 강도저하가 심하다. 
			\end{enumerate}
		\end{columns}

			\item 열처리강(High-strength quenched and tempered alloy steel)

		\begin{alertblock}{강재를 구성하는 주요원소(표2.2에서 발췌)}
			\begin{description}
				\item[철(Fe)] 함유량 98\% 이상. 강재의 대부분을 차지하는 구성요소.
				\item[탄소(C)] 함유량 0.04-2\%. 강재에서 철 다음으로 중요한 요소이며 탄소량이 증가하면 강도는 증가하나 연성이나 용접성이 떨어짐. $\Rightarrow$ 강재에서 탄소량은 강재의 성질에 결정적인 영향을 미침.
			\end{description}
			
		\end{alertblock}

	\begin{frame}{5.4 강재의 기계적 성질}
		\textbf{5.4.2 반복 및 충격하중에 대한 강재의 특성}
		\begin{description}
			\item[취성파괴(brittle failure)] 충격하중에 의하여 \alert{부재가 갑작스럽게 파괴되는 현상.} \alert{1) 주위 온도의 저하로 인해 부재의 인성이 감소되어 에너지 흡수능력이 저하되는 경우}와 \alert{2) 하중의 갑작스러운 집중(충격)현상}이 외적 요인이다. \alert{3) 부재의 가공이나 접합 시의 결함으로 생긴 균열} 또한 원인이 될 수 있다. \alert{\textbf{[참고: 연성파괴(ductile failure)]}}
			\item[에너지 흡수] 강재의 변형이 탄성을 넘어 소성으로 진행될 경우 외부힘을 제거하더라도 소성변형은 남게 된다. 이러한 강재의 소성변형 과정에서 에너지 흡수가 발생하게 되며 대부분 열로 변환된다. \alert{강재의 에너지 흡수 능력을 유효하게 사용하는 것이 내진설계 상 유리한 개념이다. }
			\item[피로(fatigue failure)] 피로현상은 \alert{구조체가 항복강도 이하의 강도를 유발하는 반복하중을 장기간 받을 때, 균열이 심화되는 경우}를 가리킨다. 금속 조직의 불연속으로 인한 응력집중은 미세한 균열에도 불구하고 장기간 하중이 작용할 경우 균열이 증가하게 된다. 피로는 주변 환경과 하중재하방법, 부재의 형태에 영향을 받는다. 
		\end{description}
	\end{frame}
