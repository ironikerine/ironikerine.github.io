\section{4. 철근콘크리트구조}	
	\begin{frame}{4.1 개요}
		\textbf{1. 철근콘크리트의 원리}
		
		\textbf{철근콘크리트(reinforced concrete)} \alert{철근}의 주위에 \alert{콘크리트}를 타설하여 일체화한 것	
		\begin{itemize}
			\item 콘크리트: 시멘트, 모래, 자갈 등의 혼합물에 물을 첨가하여 유동체로 만든 다음 이것을 일정한 형태로 굳힌 것. 압축강도는 20--120 MPa이고 인장강도는 압축강도의 5 -- 10\%임. 일반적으로 인장강도는 휨 설계 시 무시. 
			\item 강재: 높은 강도를 갖는 우수한 구조재료이지만 고가이며 화열이나 녹에 취약하다는 결점이 있음. 인장강도는 240--600 MPa임(SD240, SD500 and SD600).
			\item 콘크리트의 약한 인장강도를 철근이 보완.
			\item 콘크리트 내화피복을 통해 철근의 낮은 내화성능 보완.
		\end{itemize}
	\end{frame}
	\begin{frame}{4.1 개요}
		\textbf{2. 철근콘크리트의 장단점}
		
		\alert{장점}
		
		\begin{itemize}
			\item 공사비와 유지비가 저렴함.
			\item 내구성과 내화학성이 뛰어남.
			\item 다양하고 자유로운 조형이 가능함. 
		\end{itemize}
		
		\alert{단점}
		
		\begin{itemize}
			\item 무게에 비하여 강도가 낮음. 비중은 철골에 비해 낮으나(1/3수준), 강도가 더욱 낮기 때문에(1/10수준) 철골에 비해 긴 경간에서는 불리. 
			\item 인장강도가 압축강도의 10\%수준이므로, 인장 측에서의 균열이 쉽게 발생함. 
			\item 거푸집과 지주(동바리)의 사용: 타설 이후에도 일정기간이 지난 후에 강도가 발현되기 때문에 거푸집을 지주로 지지해야 함. 전체 공사비의 30\% 정도. 
			\item 시간에 따른 변형을 일으킴[크리프(creep): 하중과 관계있는 시간경과에 따른 변형; 건조수축(shrinkage): 하중과 관계없는 시간경과에 따른 변형].
			\item 시공이 번거로움. 
		\end{itemize}
	\end{frame}
	\begin{frame}{4.1 개요}
		\textbf{3. 철근콘크리트구조의 종류}
		
		\alert{1) 구조방식에 의한 분류}
		
		\begin{description}
			\item[라멘구조] 기둥과 보를 강접합으로 처리하여 수평하중을 지탱함. 가장 일반적인 RC조의 구법.
			\item[벽식구조] 벽체와 슬래브로 구성된 구조. 기둥이나 보가 없으며 수평하중은 물론 연직하중도 내력벽으로 지탱함. 공동주택 등 많은 벽체가 필요하고 낮은 층고가 허용되는 구조에 적합함. 
			\item[무량판구조] 보를 사용하지 않고 기둥이 직접 바닥슬래브를 지지하는 방식. 시공이 매우 편리하며 보가 없으므로 층고를 낮출 수 있음. 공동주택에 주로 사용.
			\item[쉘구조] 지붕이나 벽을 3차원의 절판이나 곡면판으로 처리하여 면외방향의 강성을 얻도록 한 것. 대공간을 구성하는 구조이나 현재에는 많이 사용되지 않음. 
		\end{description}
	\end{frame}

	\begin{frame}{4.2 재료}
		\textbf{1. 재료의 구성}
		
		보통강도 일반 콘크리트: 시멘트, 물, 모래(세골재), 자갈(조골재), 혼화재
		\begin{itemize}
			\item 무게비율 = 1:0.5:2:4
			\item 강도, 시공연도(workability), 크리프 등 요구특성에 따라서 혼합비율이 달라짐
			\item 고강도 콘크리트 사용 시: 물/시멘트 비율이 0.3까지 낮아짐. 조골재의 크기를 감소시킴(직경 25mm 이하).
			\item 콘크리트펌프 이용 시 세골재의 비율이 높아짐.
		\end{itemize}
		골재의 요구조건
		\begin{itemize}
			\item 충분한 강도, 깨끗한 표면.
			\item 세골재와 조골재의 적절한 크기의 비율(조밀한 구성).
			\item 콘크리트 타설에 적절한 최대골재크기(거푸집과 철근 사이에 밀실한 타설).
		\end{itemize}
		수화작용, 양생(curing): 시멘트는 수경성이므로 지속적인 물의 공급이 중요함. 
	\end{frame}
	\begin{frame}{4.2 재료}
		\textbf{2. 시멘트의 종류}
		\begin{description}
			\item[1종] 보통시멘트
			\item[2종] 중용열시멘트: 시공연도 증가, 동파에 의한 내구성 증진. 도로포장, 노출콘크리트에 사용. 수화작용의 속도가 빠르지 않아서 발열량이 적고 장기강도가 뛰어남. 
			\item[3종] 조강시멘트: 빠른 시공이 요구될 때 사용. 동절기에 사용. 분말도가 큼.
			\item[4종] 저열시멘트: mass concrete에 사용(대규모의 기초판 등). Mass가 큰 경우 내부와 외부의 수화열발산정도의 차이에 의하여 균열 발생. 
			\item[5종] 내황산염시멘트: 공장, 오수처리시설에 사용(일반시멘트는 황산염에 매우 약함). 
		\end{description}
	\end{frame}
	\begin{frame}{4.2 재료}
		\textbf{3. 혼화재(admixture)의 종류}
		
		강도증진, 내구성증진, 조기강도발현(조강제), 강도발현지연(완결제), 시공연도증진(감수제, 표면활성제), 내화학성증진 등 다양한 혼화제 사용 가능.
		\begin{itemize}
			\item \textbf{Fly ash, pozzolan} 내구성 증진, 조직을 치밀하게 만듬.
			\item \textbf{Silica fume} 고강도 콘크리트에 사용.
			\item \textbf{Superplasticizer} 시공연도 증진, 콘크리트의 압송에 필수적임. 			
		\end{itemize}
	\end{frame}
	\begin{frame}{4.2 재료}
		\textbf{4. 콘크리트의 종류}
		
		\begin{itemize}
			\item \textbf{보통중량콘크리트} 무게 2.4 ton/m$^3$.
			\item \textbf{경량콘크리트} 구조용, 단열용, 경량골재사용. 무게 1.0--1.8 ton/m$^3$. 구조용으로는 주로 슬래브에 사용. 
			\item \textbf{고강도콘크리트} 80 MPa 까지는 일반시멘트로 생산 가능. 100--150 MPa 는 혼화재(silica fume, 슬래그 등) 사용. 
			\item \textbf{유동화콘크리트} 유동화제(plasticizer)를 첨가하여 시공성 증진. 
		\end{itemize}
		\textbf{5. 물--시멘트 비}
		
		시공성을 위하여 수화작용에 필요한 양 이상의 물이 필요함. 콘크리트의 quality를 위하여 60\% 이하로 유지하여야 함. 
		
		\textbf{6. 슬럼프 값}
		
		콘크리트의 시공연도를 판단하는 지표. 표준 슬럼프 값: 기둥, 벽(100--150 mm); 바닥, 보(50--100 mm).	
	\end{frame}

			\item[이형철근] 부착력의 증대를 위하여 마디를 형성하고 있음. 
			\item[Epoxy-coated bar] 부식이 우려되는 곳에 사용. 교량/주차장. 
		\end{description}

		\textbf{3. 주철근의 배치} 
		
		\alert{1) 주근의 위치}
		
		\textbf{주근(longitudinal bars)} 기둥, 보 부재의 축방향으로 배치되어 축력과 휨모멘트를 저항하는 철근. 철근배근의 형태를 갖추기 위하여 설치하는 경우도 있으며, 온도팽창이나 건조수축에 의한 균열을 제어하기 위해 배치하기도 함. 
		\begin{description}
			\item[보] 주로 휨모멘트에 의하여 인장력이 발생하는 위치에 주근을 배치.
			\item[기둥] 축력과 휨모멘트를 저항하도록 주근을 배치함. 일반적으로 단면에 고르게 분포하도록 배치하며, 동일한 배치를 전체 길이에 걸쳐서 배치함. 
		\end{description}
