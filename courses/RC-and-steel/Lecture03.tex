	\begin{frame}{2.2 내력, 응력과 변형}
		\textbf{자유물체도(free body diagram)}: 물체에 작용하는 \alert{외력(하중과 반력)}을 도식적으로 표현한 그림.
		
		\begin{itemize}
		 	\item \alert{정적평형을 이룬 경우, 모든 외력의 합은 0 (개념적으로 자명함)}.
		 	\item \alert{반드시 전체 구조물이 도식화되어야 하는 것은 아님}. 구조물을 \alert{절단}하여 절단된 부재에 가해지는 \alert{외력}을 얻을 수 있고, 이 부재에 가해지는 외력으로부터 부재에 발생할 저항력인 \alert{내력}을 구하여 \alert{``구조를 해석''}함. 
		\end{itemize}

		\textbf{A} (a 지점) $V_a=R_1=P/4, M_a=-PL/8$
		
		\begin{description}
			\item[$\Sigma F_y=0$] $\rightarrow R_1 - V_a=0$
			\item[$\Sigma M=0$] (at a) $\rightarrow R_1\times L/2 + M_a=0$
		\end{description}

			\item[$\Sigma M=0$] (at b) $\rightarrow R_1\times 3L/4 + M_b=0$

	\begin{frame}{예제02: 보의 전단력과 모멘트선도 작성}
		\textbf{A}
		
		($0\leq x \leq 3L/4$) $V=R_1=3P/4, M=(P/4)x$
		
		\begin{description}
			\item[$\Sigma F_y=0$] $\rightarrow R_1 - V=0$
			\item[$\Sigma M=0$] (at $x$) $\rightarrow R_1\times x + M=0$
		\end{description}
		
		($3L/4\leq x \leq L$) $V=-R_2=-3P/4=3P/4, M=3/4P(L-x)$

		\begin{description}
			\item[$\Sigma F_y=0$] $\rightarrow R_2 + V=0$
			\item[$\Sigma M=0$] (at d) $\rightarrow M - R_2\times (L-x) =0$
		\end{description}
