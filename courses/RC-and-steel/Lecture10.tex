				\end{figure}

			\begin{column}{.59\textwidth}
				\begin{itemize}
					\item 건물에너지효율등극인증제도나 친환경주택에서 구분하는 국내 기후지역은 총 세 개: 중부지역, 남부지역과 제주지역
					\item 서울, 경기도, 충청북도, 강원도 지역 대부분이 중부지역 기후존에 위치
					\item 강원도의 경우 태백산맥 동쪽 지역이 겨울철 북서풍을 피하고 동해에 면하고 있어 상대적으로 따뜻$\rightarrow$남부지역 기후
					\item 충청북도와 천안의 경우 동일 위도 대비 내륙에 위치하여 타지역에 비해 추움$\rightarrow$중부지역 기후
					\item 경상북도 청송군은 지리적 위치에 기인하여 중부지역 기후
				\end{itemize}
			\end{column}
		\end{columns}

	\begin{frame}{6.2 자연형 태양열 시스템}
		\begin{figure}
			\includegraphics[height=0.35\textheight]{10SOLAR01}
			\caption{태양열 시스템 구성요소}
		\end{figure}
		\textbf{태양열 시스템의 특징}
		\begin{itemize}
			\item 태양에너지는 무한정하고 공해가 없으며, 어느 곳에서나 쉽게 적용할 수 있음
			\item 가장 유망한 대체에너지원
			\item 건물에서는 산업/수송에서와 같이 고효율의 에너지가 요구되는 것이 아니라, 냉난방이나 급탕과 같은 저온의 에너지가 사용되므로 그 활용성이 높음
		\end{itemize}
		\textbf{태양열 시스템의 분류} 자연형/설비형
		
		\textbf{태양열 시스템의 구성} 집열부/축열부/이용부/제어부
	\end{frame}
	\begin{frame}{6.2 자연형 태양열 시스템}
		\textbf{자연형 태양열 시스템의 특징}
		\begin{itemize}
			\item 각 시스템 구성부(집열부, 축열부, 이용부) 간의 에너지 전달방법이 자연순환방식에 따름
			\item 자연형 태양열 시스템은 주로 겨울의 난방을 위한 것으로, 여름철의 과열을 방지하고 냉방효과를 도모할 수 있는 방안이 동시에 강구되어야 함
			\item 설비형 시스템에 비해 경제성이 높고 신뢰도가 우수하며, 수명이 반영구적이고 관리가 용이하며, 특히 건축의 다른 디자인 요소와 균형적 조화를 이룰 수 있음
		\end{itemize}
		\textbf{자연형 태양열 시스템의 분류}
		\begin{itemize}
			\item 적용방법상의 분류: 직접획득형/간접획득형/분리획득형
			\item 물리적인 분류: 직접획득방식/축열벽방식/축열지붕방식/부착온실방식/자연대류방식/혼합형
		\end{itemize}
	\end{frame}
	\begin{frame}{직접획득방식}
		\begin{itemize}
			\item 가장 보편적인 방법으로서 태양의 복사열이 생활 공간을 통과해 축열체에 저장됨
			\item 따라서 생활공간은 집열부/축열부/이용부의 역할을 동시에 하게 됨

			\item 하절기 과열을 막기 위한 차양의 설치, 맞통풍 유도 등의 고려가 있어야 함

	\begin{frame}{축열벽(Trombe wall) 방식}
		\begin{columns}
			\begin{column}{.5\textwidth}
				\begin{itemize}
					\item 간접획득 방식의 일종으로, 축열벽의 축열체가 받아들인 태양열 에너지를 저장하여 생활공간으로 전달
					\item 실내 온도변화는 축열벽 두께에 의해 좌우되며, 축열벽 상하부에 설치된 통기구를 통해 가열된 공기를 실내로 유입시킴
				\end{itemize}
		\textbf{장점}
		\begin{enumerate}
			\item 거주공간 내 온도변화가 적음
			\item 축열된 복사에너지를 야간에 방출
			\item 비교적 추운 기후에 유리
		\end{enumerate}
		
		\textbf{단점}
		\begin{enumerate}
			\item 남측 벽의 일면은 투과체로, 다른 면은 축열체로 이중면이어야 함
			\item 벽의 부피가 크고 고가이며 조망이 결핍
		\end{enumerate}

	\begin{frame}{부착온실(Attached sun space) 방식}
		\begin{columns}
			\begin{column}{.5\textwidth}
				\begin{itemize}
					\item 거주공간과 분리된 별개의 공간에 태양복사 에너지를 받아들여 저장해 두었다가 다음에 분배할 수 있도록 하는 방식
					\item 온실 내로 유입되는 태양열의 대부분이 활용되어 겨울철의 효율은 약 65~75\%, 인접 공간에는 약 10~30\%의 열이 공급됨
				\end{itemize}
		\textbf{장점}
		\begin{enumerate}
			\item 완충지대의 역할을 하여 건물의 열손실을 줄임
			\item 자연과 가까이 할 수 있으며 기존 건물에 쉽게 적용됨
		\end{enumerate}
		
		\textbf{단점}
		\begin{enumerate}
			\item 디자인에 따라 열성능이 크게 다름
			\item 상업적인 가치가 있도록 잘 시공하려면 시공비가 비쌈
		\end{enumerate}

	\begin{frame}{축열지붕/지붕연못(Roof pond)방식}
		\begin{itemize}
			\item 지붕 자체가 집열기의 역할을 하며, 단층 지붕에만 적용이 가능하므로 복층 건물의 경우 최상층에만 적용 가능

			\item 북위 36도 이하의 낮은 위도의 지역에만 가능하며, 밤과 낮의 일교차가 심한 지역일수록 유리
		\end{itemize}
