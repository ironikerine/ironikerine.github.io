	\begin{frame}{6.2 합성구조의 특성}
		\textbf{SRC구조의 장점}
		
		RC구조 대비
		\begin{enumerate}
			\item 전단파괴와 압축파괴가 생겨도 철골의 존재로 인성이 발휘
			\item 동일단면에 다량의 철골을 배치하는 것이 가능하므로, 내력과 변형능력 확보가능
			\item 장스팬이 가능. 크리프에 의한 처짐이 적음
			\item 연직하중을 지지하는 것이 가능하며 역타공법이 가능
		\end{enumerate}
		
		철골구조 대비
		\begin{enumerate}
			\item 콘크리트가 구조 및 내화피복의 기능을 하기에 경제적
			\item 콘크리트의 강성이 강재의 강성에 추가되므로 전체적으로 큰 강성을 갖는 구조가 됨
			\item 작용하중에 대해서 수평 또는 연직변형을 작게하는 것이 가능
			\item 콘크리트에 의해서 강재의 좌굴을 억제하고, 변형성능이 큰 구조가 얻어짐
			\item 철근콘크리트에 의해 진동감쇠 효과
		\end{enumerate}
	\end{frame}
	\begin{frame}{6.2 합성구조의 특성}
		\textbf{SRC구조의 단점}
		
		RC구조 대비
		\begin{enumerate}
			\item 철골과 철근이 동일단면 내에 배치되므로 콘크리트 타설 시 혼잡한 상황이 발생
			\item 철골의 단가가 철근에 비해 높으므로 공사비가 RC조에 비해 높음
		\end{enumerate}
		
		철골구조 대비
		\begin{enumerate}
			\item 철골공사와 RC공사를 동시에 필요로 하기 때문에 시공이 복잡하고 건설비가 높음
			\item RC구조의 중량이 크기 때문에 건축물의 총중량이 철골조에 비하여 큼
		\end{enumerate}
	\end{frame}
	\begin{frame}{6.3 SRC 적용조건}
		\begin{enumerate}
			\item 중/고층 건축물에 하층부의 응력이 크게 되어, RC구조에서는 기둥재의 변형성능을 충분히 확보하기 힘든 경우
			\item 스팬이 커서 기둥 하나가 부담하는 면적이 크게 되어, RC구조에서는 기둥의 단면적이 너무 큰 경우나 보의 설계가 곤란한 경우
			\item 동일층에 강성이 다른 기둥이 혼재하여 전단력 부담의 불균형이 생기는 경우
			\item 건물의 편심이 커서 기둥재의 인성을 확보하고 싶은 경우
			\item 상부층을 강구조, 기초를 포함하는 하부층을 RC구조로 하는 경우에 중간층을 SRC구조로 함에 따라 강구조로부터 SRC구조에 응력전달을 원활하게 할 수 있음
		\end{enumerate}
	\end{frame}
	\begin{frame}{6.4 SRC구조의 부재산정 기본방침}
		\begin{enumerate}
			\item 철골부분은 좌굴하지 않게 한다
			\item 철근콘크리트부분은 RC조 설계규준에 따른다
			\item 철골부분을 이와 동등한 단면적을 가지는 철근으로 간주해도 무방하다
		\end{enumerate}

	\end{frame}

		\textbf{CFT} 콘크리트충전 강관구조

			\item 강관과 콘크리트의 장점을 최대한 활용

		\end{enumerate}
